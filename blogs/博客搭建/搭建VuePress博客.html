    # 2. 首页向下滚动

    在项目根目录下README.md中配置样式和脚本

    <style>
.anchor-down {
  display: block;
  margin: 12rem auto 0;
  width: 20px;
  height: 20px;
  font-size: 34px;
  text-align: center;
  animation: bounce-in 5s 3s infinite;
  position: absolute;
  left: 50%;
  bottom: 30%;
  margin-left: -10px;
  cursor: pointer;
}

@-webkit-keyframes bounce-in {
  0% {
    transform: translateY(0)
  }
  20% {
    transform: translateY(0)
  }
  50% {
    transform: translateY(-20px)
  }
  80% {
    transform: translateY(0)
  }
  to {
    transform: translateY(0)
  }
}

.anchor-down::before {
  content: "";
  width: 20px;
  height: 20px;
  display: block;
  border-right: 3px solid #fff;
  border-top: 3px solid #fff;
  transform: rotate(135deg);
  position: absolute;
  bottom: 10px;
}

.anchor-down::after {
  content: "";
  width: 20px;
  height: 20px;
  display: block;
  border-right: 3px solid #fff;
  border-top: 3px solid #fff;
  transform: rotate(135deg);
}
</style>

<script>
export default {
  mounted() {
    const ifJanchor = document.getElementById("JanchorDown");
    ifJanchor && ifJanchor.parentNode.removeChild(ifJanchor);
    let a = document.createElement('a');
    a.id = 'JanchorDown';
    a.className = 'anchor-down';
    document.getElementsByClassName('hero')[0].append(a);
    let targetA = document.getElementById("JanchorDown");
    targetA.addEventListener('click', e => { // 添加点击事件
      this.scrollFn();
    })
  },

  methods: {
    scrollFn() {
      const windowH = document.getElementsByClassName('hero')[0].clientHeight; // 获取窗口高度
      document.documentElement.scrollTop = windowH; // 滚动条滚动到指定位置
    }
  }
}
</script>

    # 二、配置

    # 1. 设置中文

    在项目根目录下.vuepress/config.js中配置locales

    module.exports = {
    locales: {
        // 默认显示中文
        '/': {
            lang: 'zh-CN'
        }

    # 三、插件

    # 1. 自动侧边栏

    • 安装依赖
    npm install vuepress-plugin-auto-sidebar
    1
    • 进行配置

    在项目根目录下.vuepress/config.js中配置plugins

    module.exports = {
    plugins: [
        [
            // 自动侧边栏
            // 因使用vuepress-theme-reco主题，可在主题配置themeConfig中去除sidebar属性
            "vuepress-plugin-auto-sidebar",
            {
                collapse: {
                    // true 左侧侧边栏，显示展开/收起
                    open: true,
                }
            }
        ]
    ]
}

    # 2. 鼠标点击特效

    • 安装依赖
    npm install vuepress-plugin-cursor-effects
    1
    • 进行配置

    在项目根目录下.vuepress/config.js中配置plugins

    module.exports = {
    plugins: [
        [
            // 鼠标点击特效
            "vuepress-plugin-cursor-effects",
            {
                size: 2,                    // 粒子大小，默认：2
                shape: 'circle',  // 粒子的形状，默认值：'star'
                zIndex: 999999999           // 画布属性，默认值： 999999999
            }
        ]
    ]
}
